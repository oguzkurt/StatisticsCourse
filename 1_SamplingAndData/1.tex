def myperc(data,p):
    n=len(data)
    if 0<=p and p<=100:
        l=p*(n+1)/100
        f=l.floor()
        r=l-f
        if f<n:
            new=data[f-1]+r*(data[f]-data[f-1])
            show(new)
            return new
        else:
            return data[n-1]
    else:
